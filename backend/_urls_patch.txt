from django.urls import path, include
from rest_framework import routers
from reception.views import PatientViewSet, AppointmentViewSet
from scheduling.views import availability
from kpi.views import overview as kpi_overview
from cashbox.views import close_session
from reports.views import export_billing_csv

router = routers.DefaultRouter()
router.register(r'patients', PatientViewSet)
router.register(r'appointments', AppointmentViewSet)

urlpatterns += [
    path('api/', include(router.urls)),
    path('api/availability/slots/', availability),
    path('api/kpi/overview/', kpi_overview),
    path('api/cash/close-session/', close_session),
    path('api/reports/billing.csv', export_billing_csv),
]
